## Simple Example

<Playground>
  {() => {
    const initial =
      "Yo {{ NAME }},\n\nThank you for your Biz! We made poor use of the money.\n\nLove,\nBusiness";

    const [text, setText] = useState(initial);

    return <>
      <InputTemplate value={text} onChange={handleChange} />
      <Button label="Reset" onClick={resetText} />
    </>;

    function handleChange(newText) {
      console.log("handleChange");
      setText(newText);
    }

    function resetText() {
      setText(initial);
    }

}}

</Playground>
